```typescript
// composables/usePreferences.ts

import { ref, computed, watch } from 'vue'
import { useStorage } from '@vueuse/core'
import { preferencesApi } from '@/services/api'
import type { UserPreferences, ViewType, AccentColor } from '@/types'

const defaultPreferences: UserPreferences = {
  appearance: {
    theme: 'system',
    accentColor: 'blue',
    fontSize: 'medium',
    reducedMotion: false,
    highContrast: false
  },
  navigation: {
    defaultView: 'tiles',
    sidebarCollapsed: false,
    enableAnimations: true,
    enableHoverEffects: true,
    showModuleBadges: true
  },
  dashboard: {
    startPage: '/dashboard',
    visibleWidgets: ['active_audits', 'appointments', 'tasks', 'activities'],
    widgetOrder: ['active_audits', 'appointments', 'tasks', 'activities'],
    compactMode: false
  },
  locale: {
    language: 'de',
    dateFormat: 'DD.MM.YYYY',
    timeFormat: '24h',
    timezone: 'Europe/Berlin',
    numberFormat: 'de-DE'
  },
  // ... rest
}

export function usePreferences() {
  // Local storage fÃ¼r schnellen Zugriff
  const localPrefs = useStorage<UserPreferences>(
    'flowaudit-preferences',
    defaultPreferences
  )

  const isLoading = ref(false)
  const isSaving = ref(false)

  // Computed Helpers
  const isDark = computed(() => {
    if (localPrefs.value.appearance.theme === 'system') {
      return window.matchMedia('(prefers-color-scheme: dark)').matches
    }
    return localPrefs.value.appearance.theme === 'dark'
  })

  const currentView = computed(() => localPrefs.value.navigation.defaultView)
  const accentColor = computed(() => localPrefs.value.appearance.accentColor)

  // Theme auf Document anwenden
  watch(isDark, (dark) => {
    document.documentElement.classList.toggle('dark', dark)
  }, { immediate: true })

  // Akzentfarbe anwenden
  watch(accentColor, (color) => {
    document.documentElement.setAttribute('data-accent', color)
  }, { immediate: true })

  // Vom Server laden
  async function loadFromServer() {
    isLoading.value = true
    try {
      const serverPrefs = await preferencesApi.get()
      if (serverPrefs) {
        localPrefs.value = { ...defaultPreferences, ...serverPrefs }
      }
    } finally {
      isLoading.value = false
    }
  }

  // Zum Server speichern
  async function saveToServer() {
    isSaving.value = true
    try {
      await preferencesApi.update(localPrefs.value)
    } finally {
      isSaving.value = false
    }
  }

  // Einzelne Einstellung Ã¤ndern
  function setTheme(theme: 'light' | 'dark' | 'system') {
    localPrefs.value.appearance.theme = theme
    saveToServer()
  }

  function setView(view: ViewType) {
    localPrefs.value.navigation.defaultView = view
    saveToServer()
  }

  function setAccentColor(color: AccentColor) {
    localPrefs.value.appearance.accentColor = color
    saveToServer()
  }

  function setLanguage(lang: 'de' | 'en') {
    localPrefs.value.locale.language = lang
    saveToServer()
  }

  function toggleSidebar() {
    localPrefs.value.navigation.sidebarCollapsed =
      !localPrefs.value.navigation.sidebarCollapsed
    saveToServer()
  }

  // ZurÃ¼cksetzen
  function resetToDefaults() {
    localPrefs.value = { ...defaultPreferences }
    saveToServer()
  }

  return {
    preferences: localPrefs,
    isLoading,
    isSaving,

    // Computed
    isDark,
    currentView,
    accentColor,

    // Actions
    loadFromServer,
    saveToServer,
    setTheme,
    setView,
    setAccentColor,
    setLanguage,
    toggleSidebar,
    resetToDefaults
  }
}
```
